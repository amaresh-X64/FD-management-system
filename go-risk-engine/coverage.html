package services

import (
        "fd-shield/go-risk-engine/models"
        "fmt"
        "math"
        "time"
)

func quarterKey(t time.Time) string {
        q := (int(t.Month())-1)/3 + 1
        return fmt.Sprintf("%d-Q%d", t.Year(), q)
}

// AnalyzeRisk runs all 3 calculations concurrently using goroutines + channels
func AnalyzeRisk(req models.RiskRequest) models.RiskResult {
        liquidityCh := make(chan float64, 1)
        spreadCh := make(chan float64, 1)
        penaltyCh := make(chan float64, 1)

        go func() { liquidityCh <- calcLiquidityScore(req) }()
        go func() { spreadCh <- calcMaturitySpreadScore(req) }()
        go func() { penaltyCh <- calcPenaltyExposure(req) }()

        return models.RiskResult{
                LiquidityScore:      <-liquidityCh,
                MaturitySpreadScore: <-spreadCh,
                PenaltyExposure:     <-penaltyCh,
        }
}

// ── Liquidity Score (0-100) ───────────────────────────────────────────────
// How many months of expenses can liquid funds cover?
// Score = 100 if liquid funds >= 6 months expenses, scales down proportionally
func calcLiquidityScore(req models.RiskRequest) float64 {
        if len(req.Fds) == 0 {
                return 0
        }

        var shortTermPrincipal float64
        for _, fd := range req.Fds {
                if fd.FdType == "SHORT_TERM" {
                        shortTermPrincipal += fd.Principal
                }
        }

        sixMonthsExpenses := req.MonthlyExpenses * 6
        if sixMonthsExpenses == 0 {
                return 100
        }

        ratio := shortTermPrincipal / sixMonthsExpenses
        score := math.Min(ratio*100, 100)
        return round(score, 2)
}

// ── Maturity Spread Score (0-100) ─────────────────────────────────────────
// How well are FD maturities spread across time?
// Good: FDs maturing in different quarters. Bad: all maturing in same month.
func calcMaturitySpreadScore(req models.RiskRequest) float64 {
        if len(req.Fds) <= 1 {
                // Single FD = worst spread
                return 20
        }

        quarters := make(map[string]int)
        for _, fd := range req.Fds {
                t, err := time.Parse("2006-01-02", fd.MaturityDate)
                if err != nil {
                        continue
                }
                q := quarterKey(t)
                quarters[q]++
        }

        // More distinct quarters = better spread
        distinctQuarters := len(quarters)
        totalFds := len(req.Fds)

        // Max score when every FD is in a different quarter
        ratio := float64(distinctQuarters) / float64(totalFds)
        score := ratio * 100
        return round(score, 2)
}

// ── Penalty Exposure (0-100) ──────────────────────────────────────────────
// Probability of needing to break FDs based on expense ratio.
// High expenses relative to income + lots of long-term FDs = high risk.
func calcPenaltyExposure(req models.RiskRequest) float64 {
        if req.MonthlyIncome == 0 {
                return 100
        }

        expenseRatio := req.MonthlyExpenses / req.MonthlyIncome

        var longTermPrincipal, totalPrincipal float64
        for _, fd := range req.Fds {
                totalPrincipal += fd.Principal
                if fd.FdType == "LONG_TERM" {
                        longTermPrincipal += fd.Principal
                }
        }

        var longTermRatio float64
        if totalPrincipal > 0 {
                longTermRatio = longTermPrincipal / totalPrincipal
        }

        // Combined risk: high expense ratio + lots of locked money = high penalty risk
        riskScore := (expenseRatio*0.5 + longTermRatio*0.5) * 100
        return round(math.Min(riskScore, 100), 2)
}

// ── Helpers ───────────────────────────────────────────────────────────────

func round(val float64, places int) float64 {
        shift := math.Pow(10, float64(places))
        return math.Round(val*shift) / shift
}
